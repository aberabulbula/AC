import requests
import pandas as pd
from bs4 import BeautifulSoup

# URL of the webpage to scrape Tesla's revenue data
url = "https://www.macrotrends.net/stocks/charts/TSLA/tesla/revenue"

# Send a request to the website and get the HTML
response = requests.get(url)
soup = BeautifulSoup(response.text, 'lxml')

# Find the table containing the revenue data
tables = soup.find_all('table')

# Select the correct table (assuming the first table contains the revenue data)
tesla_revenue_table = tables[0]

# Extract the table rows
rows = tesla_revenue_table.find_all('tr')

# Prepare lists to hold the date and revenue data
dates = []
revenues = []

# Loop through rows and extract the data for row in rows[1:]: # Skip the header row
cols = row.find_all('td')
date = cols[0].text
revenue = cols[1].text.replace('$', '').replace(',', '').strip()
dates.append(date)
revenues.append(revenue)

# Create a DataFrame
tesla_revenue = pd.DataFrame({
'Date': dates,
'Revenue': revenues
})

# Display the last five rows of the tesla_revenue dataframe
print(tesla_revenue.tail())

    plt.figure(figsize=(12, 6))
    plt.plot(data.index, data['Close'], label='Closing Price')
    plt.title(title)
    plt.xlabel('Date')
    plt.ylabel('Price (USD)')

    plt.grid(True)
    plt.show()
